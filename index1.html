const express = require('express'); const bodyParser = require('body-parser'); const app = express(); // Configura o Express para usar o body-parser como middleware. app.use(bodyParser.urlencoded({ extended: false })); app.use(bodyParser.json()); // Cria uma rota POST para registrar o nome do usuário. app.post('/segunda_pagina.html', function (req, res) { const username = req.body.uname; console.log('Nome de usuário registrado:', username); // Aqui você pode adicionar o código para salvar o nome do usuário em um banco de dados ou arquivo. res.send('Nome de usuário registrado com sucesso!'); }); app.listen(3000, function () { console.log('Aplicação escutando na porta 3000!'); });
